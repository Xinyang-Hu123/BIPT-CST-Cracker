'''
import numpy as np
import matplotlib.pyplot as plt

x = np.arange(-3,3,0.1)  #均匀横轴分布
a = 0.5* x - 1              #直线模型
y1 = 1/(1+np.exp(-x))       #sigmoid函数
y2 = 1/(1+np.exp(-a))
#print(x)

plt.plot(x,y1,color='red')
plt.plot(x,y2,color='gray')
plt.plot(x,a)
plt.grid(True)
plt.show()
'''
